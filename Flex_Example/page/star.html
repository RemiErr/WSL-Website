    getline(cin, str1);

    for (n = 0; n < longlen; ++n) {
        if (n < str1.length()) {

        } else {
            output[0] = ' ';
        }

        if (n < str2.length()) {
            output[1] = str2[n];
        } else {
            output[1] = ' ';
        }

        cout << output[1] << output[0] << endl;
    }
    return 0;
}
